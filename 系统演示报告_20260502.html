🎬 如何验证系统运行

以下命令可在终端直接运行，逐一验证各模块

# 1. 启动 API 服务
cd D:\futures_v6\api
python -m uvicorn macro_api_server:app --port 8000 --reload

# 2. 健康检查
curl http://localhost:8000/health

# 3. 全品种信号
curl http://localhost:8000/api/macro/signal/all

# 4. RU 因子明细
curl http://localhost:8000/api/macro/signal/RU

# 5. 历史打分
curl "http://localhost:8000/api/macro/score-history/RU?days=10"

# 6. 风控状态
curl http://localhost:8000/api/trading/risk-status

# 7. 下单预检
curl -X POST http://localhost:8000/api/risk/simulate \
  -H "Content-Type: application/json" \
  -d '{"symbol":"RU2505","direction":"LONG","price":15000,"volume":1}'

# 8. 启动前端（需 Node.js）
cd D:\futures_v6\frontend\futures_trading\frontend
npm run dev
